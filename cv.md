###### Code:

```
function chessBoard(n,m) {
  let a = new Array(n)
  for(let i = 0;i < n;i++){
    a[i] = new Array(m)
    for(let j = 0;j < m;j++){
      if((i + j) % 2){
        a[i][j] = 'X'
      }else{
        a[i][j] = 'O'
      }
    }
  }
  return a
}
```
